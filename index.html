import tkinter as tk from tkinter import scrolledtext, messagebox, ttk import requests import json import threading from datetime import datetime class DeepSeekChatApp: def __init__(self, root): self.root = root self.root.title("DeepSeek Chat - Conversaciones Ilimitadas") self.root.geometry("900x700") # Configuración self.api_key = "" self.conversation_history = [] self.temperature = 0.7 self.setup_ui() self.load_config() def setup_ui(self): # Frame de configuración config_frame = tk.Frame(self.root, bg="#f0f0f0", relief=tk.RAISED, bd=1) config_frame.pack(fill=tk.X, padx=10, pady=5) tk.Label(config_frame, text="API Key:", bg="#f0f0f0").grid(row=0, column=0, padx=5, pady=5, sticky="w") self.api_key_entry = tk.Entry(config_frame, show="*", width=50) self.api_key_entry.grid(row=0, column=1, padx=5, pady=5) tk.Label(config_frame, text="Temperatura:", bg="#f0f0f0").grid(row=0, column=2, padx=5, pady=5) self.temp_var = tk.DoubleVar(value=0.7) self.temp_scale = tk.Scale(config_frame, from_=0, to=2, resolution=0.1, orient=tk.HORIZONTAL, variable=self.temp_var, length=150) self.temp_scale.grid(row=0, column=3, padx=5, pady=5) save_btn = tk.Button(config_frame, text="💾 Guardar Configuración", command=self.save_config) save_btn.grid(row=0, column=4, padx=10, pady=5) clear_btn = tk.Button(config_frame, text="🗑️ Limpiar Conversación", command=self.clear_conversation) clear_btn.grid(row=0, column=5, padx=5, pady=5) # Área de chat self.chat_area = scrolledtext.ScrolledText(self.root, wrap=tk.WORD, font=("Arial", 11)) self.chat_area.pack(fill=tk.BOTH, expand=True, padx=10, pady=5) self.chat_area.config(state=tk.DISABLED) # Frame de entrada input_frame = tk.Frame(self.root) input_frame.pack(fill=tk.X, padx=10, pady=5) self.message_entry = tk.Text(input_frame, height=3, font=("Arial", 11)) self.message_entry.pack(side=tk.LEFT, fill=tk.BOTH, expand=True) send_btn = tk.Button(input_frame, text="Enviar ➤", command=self.send_message, bg="#4CAF50", fg="white", font=("Arial", 12, "bold")) send_btn.pack(side=tk.RIGHT, padx=5) # Bind Enter key self.message_entry.bind("", lambda e: self.send_message()) # Mensaje de bienvenida self.add_message("assistant", "¡Bienvenido! Soy DeepSeek. ¿En qué puedo ayudarte?") def load_config(self): try: with open("deepseek_config.json", "r") as f: config = json.load(f) self.api_key = config.get("api_key", "") self.temperature = config.get("temperature", 0.7) self.api_key_entry.insert(0, self.api_key) self.temp_var.set(self.temperature) except FileNotFoundError: pass def save_config(self): self.api_key = self.api_key_entry.get() self.temperature = self.temp_var.get() config = { "api_key": self.api_key, "temperature": self.temperature } with open("deepseek_config.json", "w") as f: json.dump(config, f) messagebox.showinfo("Éxito", "Configuración guardada correctamente") def clear_conversation(self): self.conversation_history = [] self.chat_area.config(state=tk.NORMAL) self.chat_area.delete(1.0, tk.END) self.chat_area.config(state=tk.DISABLED) self.add_message("assistant", "Conversación reiniciada. ¿En qué puedo ayudarte?") def add_message(self, role, content): self.chat_area.config(state=tk.NORMAL) # Agregar timestamp timestamp = datetime.now().strftime("%H:%M") if role == "user": self.chat_area.insert(tk.END, f"\n{'─'*50}\n", "separator") self.chat_area.insert(tk.END, f"👤 Tú ({timestamp}):\n", "user_label") self.chat_area.insert(tk.END, f"{content}\n", "user_message") else: self.chat_area.insert(tk.END, f"\n{'─'*50}\n", "separator") self.chat_area.insert(tk.END, f"🤖 DeepSeek ({timestamp}):\n", "assistant_label") self.chat_area.insert(tk.END, f"{content}\n", "assistant_message") # Configurar tags para colores self.chat_area.tag_config("user_label", foreground="blue", font=("Arial", 10, "bold")) self.chat_area.tag_config("assistant_label", foreground="green", font=("Arial", 10, "bold")) self.chat_area.tag_config("user_message", foreground="black", font=("Arial", 11)) self.chat_area.tag_config("assistant_message", foreground="black", font=("Arial", 11)) self.chat_area.tag_config("separator", foreground="gray") self.chat_area.see(tk.END) self.chat_area.config(state=tk.DISABLED) # Guardar historial self.conversation_history.append({"role": role, "content": content}) # Limitar historial if len(self.conversation_history) > 50: self.conversation_history = self.conversation_history[-50:] def send_message(self): message = self.message_entry.get("1.0", tk.END).strip() if not message: return if not self.api_key: messagebox.showwarning("API Key requerida", "Por favor, configura tu API key en el panel superior") return # Limpiar entrada self.message_entry.delete("1.0", tk.END) # Agregar mensaje del usuario self.add_message("user", message) # Deshabilitar entrada self.message_entry.config(state=tk.DISABLED) # Enviar en hilo separado thread = threading.Thread(target=self.get_assistant_response) thread.daemon = True thread.start() def get_assistant_response(self): try: messages = [ {"role": "system", "content": "Eres un asistente útil y amigable."} ] messages.extend(self.conversation_history) response = requests.post( "https://api.deepseek.com/v1/chat/completions", headers={ "Authorization": f"Bearer {self.api_key}", "Content-Type": "application/json" }, json={ "model": "deepseek-chat", "messages": messages, "temperature": self.temperature, "max_tokens": 4096 }, timeout=30 ) if response.status_code == 200: data = response.json() assistant_message = data["choices"][0]["message"]["content"] self.root.after(0, self.add_message, "assistant", assistant_message) else: error_msg = f"Error {response.status_code}: {response.text}" self.root.after(0, self.add_message, "assistant", f"❌ {error_msg}") except Exception as e: self.root.after(0, self.add_message, "assistant", f"❌ Error: {str(e)}") finally: self.root.after(0, lambda: self.message_entry.config(state=tk.NORMAL)) self.root.after(0, lambda: self.message_entry.focus()) if __name__ == "__main__": root = tk.Tk() app = DeepSeekChatApp(root) root.mainloop()
